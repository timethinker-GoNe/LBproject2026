// FarmingEngine ↔ DialogueQuests 브릿지
#if DIALOGUE_QUESTS
using DialogueQuests;
#endif

namespace FarmingEngine {
  public class DialogueQuestsWrap : MonoBehaviour {

    // 이벤트 연결 (Awake)
    narrative.onPauseGameplay  += () => TheGame.Get().PauseScripts();
    narrative.onUnpauseGameplay+= () => TheGame.Get().UnpauseScripts();
    narrative.onPlaySFX        += (ch, clip, vol) => TheAudio.Get().PlaySFX(ch, clip, vol);
    narrative.getTimestamp     += () => TheGame.Get().GetTimestamp();   // 시간 공유

    // 저장 동기화
    the_game.beforeSave += () => NarrativeData.Save(pdata.filename, ...);

    // 로드 동기화
    TheGame.afterLoad  += () => NarrativeData.AutoLoad(pdata.filename);
    TheGame.afterNewGame += () => NarrativeData.NewGame(pdata.filename);
  }
}

    if (item.id == id) return item;

// ActionCook.cs (MAction 상속)
public class ActionCook : MAction {
    public ItemData cooked_item;
    public float duration = 0.5f;
    public float energy = 1f;

    public override void DoAction(...) {
        character.TriggerProgressBusy(duration, () => {
            inventory.RemoveItemAt(slot.index, 1);
            character.Inventory.GainItem(cooked_item, 1);
            character.Attributes.AddAttribute(
                AttributeType.Energy, -energy);
        });
    }
    public override bool CanDoAction(...) {
        return character.Attributes
            .GetAttributeValue(AttributeType.Energy) >= energy;
    }
}

// TheUI.cs Update() — 포커스 직접 관리
void Update() {
    // 게임패드 자동 포커스 — 조건 분산
    UISlotPanel focus = UISlotPanel.GetFocusedPanel();

    if (focus != pause_panel
        && TheGame.Get().IsPausedByPlayer()
        && PlayerControls.IsAnyGamePad()) {
        pause_panel.Focus();  // 직접 패널 조작
    }
    if (focus == pause_panel
        && !TheGame.Get().IsPausedByPlayer()) {
       UISlotPanel.UnfocusAll();
    }
}

    public void PushState(UIState newState) {
        state_stack.Push(current_state);
        TransitionTo(newState);
    }

    public void PopState() {
        if (state_stack.Count > 0)
            TransitionTo(state_stack.Pop());
    }

    // IsBlockingPanelOpened — 자동 계산
    public bool IsBlocking() =>
        current_state != UIState.Gameplay;
}

// SaveSlotInfo.cs — 세이브 슬롯 표시용 데이터
[System.Serializable]
public class SaveSlotInfo
{
    public string   slot_name;       // "slot1", "slot2", "slot3"
    public bool     has_data;        // 저장 파일 존재 여부

    // PlayerData (.farming)에서 읽어온 정보
    public int      day;             // 진행 일수
    public float    day_time;        // 게임 내 시각 (0~24)
    public float    play_time;       // 실제 플레이 시간(초)
    public string   current_scene;   // 마지막 씬 이름
    public DateTime last_save;       // 마지막 저장 시각
    public string   version;         // 게임 버전

    // 사람이 읽기 좋은 형태로 변환
    public string GetDayTimeString() =>
        $"Day {day} {(int)day_time:D2}:{(int)((day_time%1)*60):D2}";

    public string GetPlayTimeString() {
        int h = (int)(play_time / 3600);
        int m = (int)(play_time % 3600 / 60);
        return $"{h}h {m}m";
    }
}

// 파일 존재 여부만 확인 후 최소 데이터 로드
public static SaveSlotInfo ReadSlotInfo(string slot) {
    var info = new SaveSlotInfo();
    info.slot_name = slot;
    info.has_data = SaveTool.DoesFileExist(slot+".farming");
    if (info.has_data) {
        var pdata = SaveTool.LoadFile<PlayerData>(slot+".farming");
        info.day = pdata?.day ?? 0;
        info.play_time = pdata?.play_time ?? 0;
        info.current_scene = pdata?.current_scene ?? "";
        info.last_save = pdata?.last_save ?? DateTime.MinValue();
    }
    return info;
}

// SaveManager.cs — FarmingEngine + DialogueQuests 통합 저장 관리자
namespace FarmingEngine
{
    public class SaveManager : MonoBehaviour
    {
        public static readonly string[] SLOT_NAMES = { "slot1", "slot2", "slot3" };
        public static readonly string LAST_SLOT_KEY = "last_save_slot";

        private static SaveManager _instance;
        public static SaveManager Get() => _instance;

        // ── 슬롯 정보 읽기 (저장 선택 UI용) ─────────────────
        public SaveSlotInfo[] GetAllSlotInfos()
        {
            var infos = new SaveSlotInfo[SLOT_NAMES.Length];
            for (int i = 0; i < SLOT_NAMES.Length; i++)
                infos[i] = SaveSlotInfo.ReadSlotInfo(SLOT_NAMES[i]);
            return infos;
        }

        // ── 저장 (두 시스템 동시) ────────────────────────────
        public void SaveToSlot(int slot_index)
        {
            string slot = SLOT_NAMES[slot_index];

            // 1. FarmingEngine 저장
            TheGame.Get().Save(slot);          // → PlayerData.Save(slot)

            // 2. DialogueQuests 저장
            #if DIALOGUE_QUESTS
            var dqData = DialogueQuests.NarrativeData.Get();
            if (dqData != null)
                DialogueQuests.NarrativeData.Save(slot, dqData);
            #endif

            // 3. 마지막 슬롯 기록
            PlayerPrefs.SetString(LAST_SLOT_KEY, slot);
            PlayerPrefs.Save();

            Debug.Log($"[SaveManager] 저장 완료 → {slot}");
        }

        // ── 불러오기 (두 시스템 동시) ───────────────────────
        public void LoadFromSlot(int slot_index)
        {
            string slot = SLOT_NAMES[slot_index];
            if (!SaveTool.DoesFileExist(slot + ".farming")) return;

            // FarmingEngine 로드 (씬 전환까지 포함)
            TheGame.Load(slot);    // afterLoad 이벤트 → DQ도 자동 로드

            PlayerPrefs.SetString(LAST_SLOT_KEY, slot);
        }

        // ── 삭제 (두 파일 동시) ──────────────────────────────
        public void DeleteSlot(int slot_index)
        {
            string slot = SLOT_NAMES[slot_index];
            SaveTool.DeleteFile(slot + ".farming");
            SaveTool.DeleteFile(slot + ".dq");       // DQ 세이브도 삭제
        }

        // ── 마지막 슬롯 자동 로드 (게임 시작 시) ───────────
        public void AutoLoadLastSlot()
        {
            string slot = PlayerPrefs.GetString(LAST_SLOT_KEY, "slot1");
            if (SaveTool.DoesFileExist(slot + ".farming"))
                LoadFromSlot(System.Array.IndexOf(SLOT_NAMES, slot));
            else
                TheGame.NewGame(slot);  // 세이브 없으면 새 게임
        }
    }
}

// SaveSlotPanel.cs
public class SaveSlotPanel : UIPanel
{
    public enum SavePanelMode { Save, Load }

    [SerializeField] SaveSlotUI[] slot_uis;  // 3개
    [SerializeField] ConfirmPanel confirm;   // 확인 모달

    private SavePanelMode mode;

    public void Show(SavePanelMode mode)
    {
        this.mode = mode;
        var infos = SaveManager.Get().GetAllSlotInfos();
        for (int i = 0; i < slot_uis.Length; i++)
            slot_uis[i].Refresh(infos[i], mode);
        base.Show();
    }

    public void OnClickSlot(int index)
    {
        string msg = mode == SavePanelMode.Save
            ? $"슬롯 {index+1}에 저장하시겠습니까?"
            : $"슬롯 {index+1}을 불러오시겠습니까?";

        confirm.Show(msg, () => {
            if (mode == SavePanelMode.Save)
                SaveManager.Get().SaveToSlot(index);
            else
                SaveManager.Get().LoadFromSlot(index);
            Hide();
        });
    }

    public void OnClickDelete(int index)
    {
        confirm.Show("정말 삭제하시겠습니까?", () => {
            SaveManager.Get().DeleteSlot(index);
            Refresh();
        });
    }
}
